CREATE DATABASE nfsen

CREATE TABLE nfsen.flows
(
    `ts` DateTime DEFAULT 0,
    `te` DateTime DEFAULT 0,
    `sa` String,
    `da` String,
    `sp` UInt16 DEFAULT 0,
    `dp` UInt16 DEFAULT 0,
    `pr` Nullable(String),
    `flg` String,
    `ipkt` UInt64,
    `ibyt` UInt64,
    `ra` String,
    `flowsrc` String
)
ENGINE = MergeTree
PARTITION BY tuple()
PRIMARY KEY (ts, te)
ORDER BY (ts, te, sa, da)
TTL te + toIntervalDay(90)
